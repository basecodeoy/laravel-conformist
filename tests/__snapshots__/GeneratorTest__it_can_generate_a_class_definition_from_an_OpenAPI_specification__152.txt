<?php

declare(strict_types=1);

namespace App\Conformist\GitHub\Reactions;

use BaseCodeOy\Conformist\Contracts\Connector;
use BaseCodeOy\Conformist\Enums\HttpMethod;
use BaseCodeOy\Conformist\Request\AbstractRequest;

/**
 * @see https://docs.github.com/rest/reference/reactions#list-reactions-for-a-team-discussion-comment
 */
final class ListForTeamDiscussionCommentInOrgRequest extends AbstractRequest
{
    public function __constructor(
        private string $org,
        private string $teamSlug,
        private int $discussionNumber,
        private int $commentNumber,
        private string $content,
        private int $perPage,
        private int $page,
    ) {
        //
    }

    public function path(): string
    {
        return '/orgs/{org}/teams/{team_slug}/discussions/{discussion_number}/comments/{comment_number}/reactions';
    }

    public function connector(): Connector
    {
        return new GitHubConnector;
    }

    public function method(): HttpMethod
    {
        return HttpMethod::GET;
    }

    /** @return Extension[] */
    public function extensions(): array
    {
        return [
            new WithUrlParameters([
                'org' => $this->org,
                'team_slug' => $this->teamSlug,
                'discussion_number' => $this->discussionNumber,
                'comment_number' => $this->commentNumber,
            ]),
            new WithQueryParameters([
                'content' => $this->content,
                'per_page' => $this->perPage,
                'page' => $this->page,
            ]),
        ];
    }
}
