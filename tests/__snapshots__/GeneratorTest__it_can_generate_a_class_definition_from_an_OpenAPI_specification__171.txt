<?php

declare(strict_types=1);

namespace App\Conformist\GitHub\Projects;

use PreemStudio\Conformist\Contracts\Connector;
use PreemStudio\Conformist\Enums\HttpMethod;
use PreemStudio\Conformist\Request\AbstractRequest;

/**
 * @see https://docs.github.com/rest/reference/projects#list-project-collaborators
 */
final class ListCollaboratorsRequest extends AbstractRequest
{
    public function __constructor(
        private int $projectId,
        private string $affiliation,
        private int $perPage,
        private int $page,
    ) {
        //
    }

    public function path(): string
    {
        return '/projects/{project_id}/collaborators';
    }

    public function connector(): Connector
    {
        return new GitHubConnector;
    }

    public function method(): HttpMethod
    {
        return HttpMethod::GET;
    }

    /** @return Extension[] */
    public function extensions(): array
    {
        return [
            new WithUrlParameters([
                'project_id' => $this->projectId,
            ]),
            new WithQueryParameters([
                'affiliation' => $this->affiliation,
                'per_page' => $this->perPage,
                'page' => $this->page,
            ]),
        ];
    }
}
