<?php

declare(strict_types=1);

namespace App\Conformist\GitHub\Actions;

use BombenProdukt\Conformist\Contracts\Connector;
use BombenProdukt\Conformist\Enums\HttpMethod;
use BombenProdukt\Conformist\Request\AbstractRequest;

/**
 * @see https://docs.github.com/rest/reference/actions#list-jobs-for-a-workflow-run
 */
final class ListJobsForWorkflowRunRequest extends AbstractRequest
{
    public function __constructor(
        private string $owner,
        private string $repo,
        private int $runId,
        private string $filter,
        private int $perPage,
        private int $page,
    ) {
        //
    }

    public function path(): string
    {
        return '/repos/{owner}/{repo}/actions/runs/{run_id}/jobs';
    }

    public function connector(): Connector
    {
        return new GitHubConnector;
    }

    public function method(): HttpMethod
    {
        return HttpMethod::GET;
    }

    /** @return Extension[] */
    public function extensions(): array
    {
        return [
            new WithUrlParameters([
                'owner' => $this->owner,
                'repo' => $this->repo,
                'run_id' => $this->runId,
            ]),
            new WithQueryParameters([
                'filter' => $this->filter,
                'per_page' => $this->perPage,
                'page' => $this->page,
            ]),
        ];
    }
}
