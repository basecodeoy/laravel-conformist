<?php

declare(strict_types=1);

namespace App\Conformist\GitHub\Actions;

use BaseCodeOy\Conformist\Contracts\Connector;
use BaseCodeOy\Conformist\Enums\HttpMethod;
use BaseCodeOy\Conformist\Request\AbstractRequest;

/**
 * @see https://docs.github.com/rest/reference/actions#disable-a-workflow
 */
final class DisableWorkflowRequest extends AbstractRequest
{
    public function __constructor(
        private string $owner,
        private string $repo,
        private int|string $workflowId,
    ) {
        //
    }

    public function path(): string
    {
        return '/repos/{owner}/{repo}/actions/workflows/{workflow_id}/disable';
    }

    public function connector(): Connector
    {
        return new GitHubConnector;
    }

    public function method(): HttpMethod
    {
        return HttpMethod::PUT;
    }

    /** @return Extension[] */
    public function extensions(): array
    {
        return [
            new WithUrlParameters([
                'owner' => $this->owner,
                'repo' => $this->repo,
                'workflow_id' => $this->workflowId,
            ]),
        ];
    }
}
