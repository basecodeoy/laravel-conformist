<?php

declare(strict_types=1);

namespace App\Conformist\GitHub\Reactions;

use BombenProdukt\Conformist\Contracts\Connector;
use BombenProdukt\Conformist\Enums\HttpMethod;
use BombenProdukt\Conformist\Request\AbstractRequest;

/**
 * @see https://docs.github.com/rest/reference/reactions#list-reactions-for-a-commit-comment
 */
final class ListForCommitCommentRequest extends AbstractRequest
{
    public function __constructor(
        private string $owner,
        private string $repo,
        private int $commentId,
        private string $content,
        private int $perPage,
        private int $page,
    ) {
        //
    }

    public function path(): string
    {
        return '/repos/{owner}/{repo}/comments/{comment_id}/reactions';
    }

    public function connector(): Connector
    {
        return new GitHubConnector;
    }

    public function method(): HttpMethod
    {
        return HttpMethod::GET;
    }

    /** @return Extension[] */
    public function extensions(): array
    {
        return [
            new WithUrlParameters([
                'owner' => $this->owner,
                'repo' => $this->repo,
                'comment_id' => $this->commentId,
            ]),
            new WithQueryParameters([
                'content' => $this->content,
                'per_page' => $this->perPage,
                'page' => $this->page,
            ]),
        ];
    }
}
