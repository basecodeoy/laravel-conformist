<?php

declare(strict_types=1);

namespace App\Conformist\GitHub\Issues;

use BaseCodeOy\Conformist\Contracts\Connector;
use BaseCodeOy\Conformist\Enums\HttpMethod;
use BaseCodeOy\Conformist\Request\AbstractRequest;

/**
 * @see https://docs.github.com/rest/reference/issues#check-if-a-user-can-be-assigned-to-a-issue
 */
final class CheckUserCanBeAssignedToIssueRequest extends AbstractRequest
{
    public function __constructor(
        private string $owner,
        private string $repo,
        private int $issueNumber,
        private string $assignee,
    ) {
        //
    }

    public function path(): string
    {
        return '/repos/{owner}/{repo}/issues/{issue_number}/assignees/{assignee}';
    }

    public function connector(): Connector
    {
        return new GitHubConnector;
    }

    public function method(): HttpMethod
    {
        return HttpMethod::GET;
    }

    /** @return Extension[] */
    public function extensions(): array
    {
        return [
            new WithUrlParameters([
                'owner' => $this->owner,
                'repo' => $this->repo,
                'issue_number' => $this->issueNumber,
                'assignee' => $this->assignee,
            ]),
        ];
    }
}
