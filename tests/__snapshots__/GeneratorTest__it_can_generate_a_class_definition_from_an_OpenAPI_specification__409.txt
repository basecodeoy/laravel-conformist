<?php

declare(strict_types=1);

namespace App\Conformist\GitHub\SecretScanning;

use BaseCodeOy\Conformist\Contracts\Connector;
use BaseCodeOy\Conformist\Enums\HttpMethod;
use BaseCodeOy\Conformist\Request\AbstractRequest;

/**
 * @see https://docs.github.com/rest/reference/secret-scanning#get-a-secret-scanning-alert
 */
final class GetAlertRequest extends AbstractRequest
{
    public function __constructor(
        private string $owner,
        private string $repo,
        private int $alertNumber,
    ) {
        //
    }

    public function path(): string
    {
        return '/repos/{owner}/{repo}/secret-scanning/alerts/{alert_number}';
    }

    public function connector(): Connector
    {
        return new GitHubConnector;
    }

    public function method(): HttpMethod
    {
        return HttpMethod::GET;
    }

    /** @return Extension[] */
    public function extensions(): array
    {
        return [
            new WithUrlParameters([
                'owner' => $this->owner,
                'repo' => $this->repo,
                'alert_number' => $this->alertNumber,
            ]),
        ];
    }
}
