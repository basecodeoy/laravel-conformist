<?php

declare(strict_types=1);

namespace App\Conformist\GitHub\Dependabot;

use BaseCodeOy\Conformist\Contracts\Connector;
use BaseCodeOy\Conformist\Enums\HttpMethod;
use BaseCodeOy\Conformist\Request\AbstractRequest;

/**
 * @see https://docs.github.com/rest/dependabot/alerts#list-dependabot-alerts-for-an-organization
 */
final class ListAlertsForOrgRequest extends AbstractRequest
{
    public function __constructor(
        private string $org,
        private string $state,
        private string $severity,
        private string $ecosystem,
        private string $package,
        private string $scope,
        private string $sort,
        private string $direction,
        private string $before,
        private string $after,
        private int $first,
        private int $last,
        private int $perPage,
    ) {
        //
    }

    public function path(): string
    {
        return '/orgs/{org}/dependabot/alerts';
    }

    public function connector(): Connector
    {
        return new GitHubConnector;
    }

    public function method(): HttpMethod
    {
        return HttpMethod::GET;
    }

    /** @return Extension[] */
    public function extensions(): array
    {
        return [
            new WithUrlParameters([
                'org' => $this->org,
            ]),
            new WithQueryParameters([
                'state' => $this->state,
                'severity' => $this->severity,
                'ecosystem' => $this->ecosystem,
                'package' => $this->package,
                'scope' => $this->scope,
                'sort' => $this->sort,
                'direction' => $this->direction,
                'before' => $this->before,
                'after' => $this->after,
                'first' => $this->first,
                'last' => $this->last,
                'per_page' => $this->perPage,
            ]),
        ];
    }
}
